1단계: 접미사 퍼지 매칭 적용 (즉시 실행)
cached_jobs의 related_majors에서 학과를 조회할 때, 정확한 이름 매칭이 안 되면 접미사 변형 매칭을 시도하는 로직을 구현해라.
매칭 규칙:
javascriptfunction fuzzyMatchMajor(majorName, cachedMajorsList) {
  // 1. 정확히 일치하면 반환
  const exact = cachedMajorsList.find(m => m.name === majorName);
  if (exact) return exact;
  
  // 2. 접미사 정규화 후 비교
  const normalize = (name) => name
    .replace(/학과$/, '')
    .replace(/과$/, '')
    .replace(/부$/, '')
    .replace(/전공$/, '');
  
  const normalized = normalize(majorName);
  return cachedMajorsList.find(m => normalize(m.name) === normalized);
}
이 매칭 로직을 다음 위치에 적용해라:

직업 카드 펼침 시 관련 전공 목록에서 "전공 상세보기" 링크 활성화 판단
