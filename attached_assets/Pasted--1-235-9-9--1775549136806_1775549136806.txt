  MAX(salary) as max_sal,
  AVG(salary) as avg_sal,
  COUNT(*) as count
FROM cached_jobs 
WHERE salary IS NOT NULL AND salary > 0;

-- 상위/하위 10개 확인
SELECT name, salary FROM cached_jobs 
WHERE salary IS NOT NULL AND salary > 0 
ORDER BY salary ASC LIMIT 10;

SELECT name, salary FROM cached_jobs 
WHERE salary IS NOT NULL AND salary > 0 
ORDER BY salary DESC LIMIT 10;
